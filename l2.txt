print("Length of Statement 1:", len(a))
print("Length of Statement 2:", len(b))

# Explanation:
# Statement 1 converts all words to lowercase BEFORE removing duplicates.
# Statement 2 removes duplicates BEFORE lowercase conversion.
# Statement 2 usually gives a larger value.

# 4. Four-letter Words in text5
# ---------------------------------------------------

four_letter_words = [
    w.lower()
    for w in text5
    if len(w) == 4 and w.isalpha()
]

fdist = FreqDist(four_letter_words)

print(fdist.most_common())

# Plot
fdist.plot(50)
